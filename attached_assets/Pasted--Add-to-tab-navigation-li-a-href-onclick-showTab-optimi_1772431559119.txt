JavaScriptfunction showTab(tabId) {
  document.querySelectorAll('.tab-content').forEach(t => t.style.display = 'none');
  document.getElementById(tabId).style.display = 'block';
}

async function runEdgeOptimizer() {
  document.getElementById('optimizer-results').innerHTML = '<p style="color:orange;">Running full optimization across all patterns, timeframes, R:R, and confluence levels... (this takes 15–25 minutes)</p>';
  
  const res = await fetch('/api/optimizer/run', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({
      months: 6,
      timeframes: ['1min','2min','3min','5min','15min'],
      patterns: 'all',
      rrRatios: [1.5, 2.0, 2.5, 3.0],
      minConfluence: 6
    })
  });
  
  const data = await res.json();
  
  let html = '<h3>Top 10 Highest Probability Setups</h3><table border="1" style="width:100%;text-align:left;">';
  html += '<tr><th>Rank</th><th>Symbol</th><th>Pattern</th><th>TF</th><th>R:R</th><th>Conf</th><th>Win%</th><th>PF</th><th>P&L</th></tr>';
  
  data.top10.forEach((row, i) => {
    html += `<tr><td>${i+1}</td><td>${row.symbol}</td><td>${row.pattern}</td><td>${row.tf}</td><td>${row.rr}</td><td>${row.confluence}</td><td>${row.winRate}%</td><td>${row.pf}</td><td>$${row.pnl}</td></tr>`;
  });
  
  html += '</table>';
  document.getElementById('optimizer-results').innerHTML = html;
}
