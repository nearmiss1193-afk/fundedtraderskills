Multi-Timeframe Signals & Convergence in Multi-Scanner
Update multi-scan endpoint in routes.ts to accept multiple timeframes and detect cross-TF alignment (signals on same bar across TFs add bonus).

JavaScriptapp.post('/api/backtest/multi', async (req, res) => {
  const { symbols, timeframes = ['5min'], patterns, minConfluence, startDate, endDate } = req.body;
  const results = [];
  const allSignals = {}; // {symbol: {timestamp: [signals across TFs]} }

  for (const tf of timeframes) {
    for (const sym of symbols) {
      const bars = await fetchPolygonAggs(sym, tf, startDate, endDate);
      const signals = detectAllPatterns(bars, sym, tf);
      const highConfluence = signals.filter(s => s.confluence >= minConfluence);

      // Store all signals for convergence check
      if (!allSignals[sym]) allSignals[sym] = {};
      highConfluence.forEach(s => {
        if (!allSignals[sym][s.timestamp]) allSignals[sym][s.timestamp] = [];
        allSignals[sym][s.timestamp].push({ ...s, tf });
      });

      results.push({ sym, tf, highConfluence });
    }
  }

  // Add convergence bonus for multi-TF alignment
  Object.keys(allSignals).forEach(sym => {
    Object.keys(allSignals[sym]).forEach(ts => {
      const sigs = allSignals[sym][ts];
      if (sigs.length >= 2) {
        const bonus = (sigs.length - 1) * 1.5;
        sigs.forEach(s => {
          s.confluence += bonus;
          s.notes = (s.notes || '') + ` Multi-TF alignment (${sigs.map(s => s.tf).join(', ')}) (+${bonus}pt)`;
        });
      }
    });
  });

  res.json({ results, allSignals });
});

function checkApexRules(signal, currentPnl, currentTime) {
  if (currentPnl <= APEX_RULES.dailyLossLimit * accountSize) {
    console.log(`Apex daily loss limit hit: ${currentPnl}`);
    return false;
  }
  if (APEX_RULES.rthOnly && !isRTH(currentTime)) {
    console.log(`Outside RTH: ${currentTime}`);
    return false;
  }
  if (signal.qty > APEX_RULES.maxContracts) {
    signal.qty = APEX_RULES.maxContracts;
    console.log(`Apex size limit enforced: qty = ${signal.qty}`);
  }
  return true;
}
