Add VWAP (Volume Weighted Average Price)
VWAP is missing from current setup. Adding it will improve mean reversion and trend strength confluence.Paste this code into patterns.ts or trader.ts:

JavaScript// VWAP Calculation - add to addIndicators() or signal processing
function calculateVWAP(bars) {
  let cumPV = 0;
  let cumVolume = 0;

  bars.forEach(bar => {
    const typicalPrice = (bar.high + bar.low + bar.close) / 3;
    cumPV += typicalPrice * bar.volume;
    cumVolume += bar.volume;
  });

  return cumPV / cumVolume;
}

// In signal processing or confluence
const vwap = calculateVWAP(currentBars.slice(-50)); // last 50 bars VWAP
const aboveVWAP = lastBar.close > vwap;
const belowVWAP = lastBar.close < vwap;

if (aboveVWAP && signal.direction === 'long') signal.confluence += 1;
if (belowVWAP && signal.direction === 'short') signal.confluence += 1;
signal.notes += ` [VWAP ${aboveVWAP ? 'Above' : 'Below'} +1]`;
