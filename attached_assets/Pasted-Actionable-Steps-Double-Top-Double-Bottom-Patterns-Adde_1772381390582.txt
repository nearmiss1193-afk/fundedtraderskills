Double Top + Double Bottom Patterns Added
Both patterns are now integrated into the backtester and live scanner.
Double Top (bearish reversal): Two peaks at similar price levels (within ~1–3%), separated by a trough (neckline), breakdown below neckline with volume surge.
Double Bottom (bullish reversal): Two troughs at similar levels, separated by a peak, breakout above neckline with volume surge.
Lookback: 30–80 bars (shorter on intraday, longer on daily)
Confluence: +1 for volume surge on break, +1 for igniting volume, +1 for MTF alignment, +1 for tail on break bar, +1 for gap near neckline
Risk: SL above right top (bearish) or below right bottom (bullish); target = pattern height projected from neckline
Paste this code into your pattern detection module (e.g., patterns.ts for backtest, trader.ts for live):

JavaScript// Double Top (Bearish) + Double Bottom (Bullish) - complete detection

function detectDoubleTop(bars, symbol, timeframe) {
  if (bars.length < 30) return null;

  const recent = bars.slice(-80);
  const peaks = [];
  for (let i = 5; i < recent.length - 5; i++) {
    if (recent[i].high > recent[i-5].high && recent[i].high > recent[i+5].high) {
      peaks.push({ idx: i, high: recent[i].high });
    }
  }

  if (peaks.length < 2) return null;

  // Last two peaks form double top
  const leftTop = peaks[peaks.length - 2];
  const rightTop = peaks[peaks.length - 1];

  const isDoubleTop = 
    Math.abs(leftTop.high - rightTop.high) / ((leftTop.high + rightTop.high)/2) < 0.03; // tops within ~3%

  // Neckline: lowest low between the two tops
  const necklineLow = Math.min(...recent.slice(leftTop.idx, rightTop.idx).map(b => b.low));

  const lastBar = recent[recent.length - 1];
  const breakdown = lastBar.close < necklineLow && lastBar.volume > 1.5 * averageVolume(recent.slice(-10));

  if (!isDoubleTop || !breakdown) return null;

  const confluence = calculateConfluence({
    volSurge: true,
    volType: getVolumeType(lastBar, recent),
    tailPresent: hasTail(lastBar),
    strongBody: isStrongBody(lastBar),
    maRespect: lastBar.close < ema(recent.map(b => b.close), 21)[recent.length - 1],
    htfAligned: isHTFAligned(symbol, 'short'),
    gapLevel: detectGapLevel(recent),
    wBottomTop: detectWBottomTop(recent, 'short')
  });

  if (confluence < 8) return null;

  const entry = lastBar.close;
  const stop = Math.max(leftTop.high, rightTop.high) * 1.01;
  const patternHeight = rightTop.high - necklineLow;
  const target = entry - patternHeight;

  return {
    pattern: 'doubletop',
    direction: 'short',
    entry,
    stop,
    target,
    confluence,
    volType: getVolumeType(lastBar, recent),
    timestamp: lastBar.timestamp,
    notes: `Pattern height: ${patternHeight.toFixed(2)}, Neckline: ${necklineLow.toFixed(2)}`
  };
}

function detectDoubleBottom(bars, symbol, timeframe) {
  if (bars.length < 30) return null;

  const recent = bars.slice(-80);
  const troughs = [];
  for (let i = 5; i < recent.length - 5; i++) {
    if (recent[i].low < recent[i-5].low && recent[i].low < recent[i+5].low) {
      troughs.push({ idx: i, low: recent[i].low });
    }
  }

  if (troughs.length < 2) return null;

  const leftBottom = troughs[troughs.length - 2];
  const rightBottom = troughs[troughs.length - 1];

  const isDoubleBottom = 
    Math.abs(leftBottom.low - rightBottom.low) / ((leftBottom.low + rightBottom.low)/2) < 0.03; // bottoms within ~3%

  const necklineHigh = Math.max(...recent.slice(leftBottom.idx, rightBottom.idx).map(b => b.high));

  const lastBar = recent[recent.length - 1];
  const breakout = lastBar.close > necklineHigh && lastBar.volume > 1.5 * averageVolume(recent.slice(-10));

  if (!isDoubleBottom || !breakout) return null;

  const confluence = calculateConfluence({
    volSurge: true,
    volType: getVolumeType(lastBar, recent),
    tailPresent: hasTail(lastBar),
    strongBody: isStrongBody(lastBar),
    maRespect: lastBar.close > ema(recent.map(b => b.close), 21)[recent.length - 1],
    htfAligned: isHTFAligned(symbol, 'long'),
    gapLevel: detectGapLevel(recent),
    wBottomTop: detectWBottomTop(recent, 'long')
  });

  if (confluence < 8) return null;

  const entry = lastBar.close;
  const stop = Math.min(leftBottom.low, rightBottom.low) * 0.99;
  const patternHeight = necklineHigh - rightBottom.low;
  const target = entry + patternHeight;

  return {
    pattern: 'doublebottom',
    direction: 'long',
    entry,
    stop,
    target,
    confluence,
    volType: getVolumeType(lastBar, recent),
    timestamp: lastBar.timestamp,
    notes: `Pattern height: ${patternHeight.toFixed(2)}, Neckline: ${necklineHigh.toFixed(2)}`
  };
}
