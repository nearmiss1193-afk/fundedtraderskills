Next Focus: Intraday Multi-Symbol Scan
Switch to 5-min timeframe for realistic Live Traders pattern frequency. Paste this updated multi-scan prompt into Replit console (server running). It uses 5-min data, 1-year range (to avoid long Polygon fetches), and min 7pt confluence.

JavaScript// Paste into Replit console and run
const axios = require('axios');

const BASE_URL = 'http://localhost:3000'; // your Replit URL

const symbols = ['ES', 'NQ', 'CL', 'GC', 'BTC', 'ETH', 'ZS']; // high-liquidity Apex set

const scanParams = {
  timeframe: '5min',
  startDate: '2025-01-01',
  endDate: '2026-02-28',
  patterns: ['3bar', '4bar', 'buysetup', 'retest', 'breakout', 'climax'],
  minConfluence: 7
};

async function runIntradayMultiScan() {
  console.log('Starting intraday multi-symbol scan...');
  console.log('Timeframe: 5min | Min Confluence: 7pt');

  try {
    const res = await axios.post(`${BASE_URL}/api/backtest/multi`, {
      symbols,
      ...scanParams
    });

    const data = res.data;

    console.log('\nSUMMARY');
    console.log(`Symbols: ${data.summary.symbolsScanned}`);
    console.log(`Total Trades: ${data.summary.totalTrades}`);
    console.log(`Win Rate: ${(data.summary.winRate * 100).toFixed(1)}%`);
    console.log(`Profit Factor: ${data.summary.profitFactor.toFixed(2)}`);
    console.log(`Net P&L: ${data.summary.netPnL.toFixed(2)}`);

    console.log('\nTOP 5 SYMBOLS BY P&L');
    data.results
      .sort((a, b) => b.totalPnL - a.totalPnL)
      .slice(0, 5)
      .forEach(r => {
        console.log(`${r.symbol.padEnd(6)} | P&L: ${r.totalPnL.toFixed(2)} | WR: ${(r.winRate * 100).toFixed(1)}% | PF: ${r.profitFactor.toFixed(2)} | Trades: ${r.tradeCount}`);
      });

    console.log('\nPATTERN BREAKDOWN');
    if (data.patternBreakdown) {
      Object.entries(data.patternBreakdown)
        .sort(([,a], [,b]) => b.pnl - a.pnl)
        .slice(0, 5)
        .forEach(([pat, s]) => {
          console.log(`${pat.padEnd(12)} | P&L: ${s.pnl.toFixed(2)} | WR: ${(s.winRate * 100).toFixed(1)}% | PF: ${s.profitFactor.toFixed(2)} | Trades: ${s.count}`);
        });
    }

    console.log('\nNext: Edge Builder → Backtest Edge Heatmap + Export CSV');

  } catch (err) {
    console.error('Error:', err.message);
  }
}

runIntradayMultiScan();
