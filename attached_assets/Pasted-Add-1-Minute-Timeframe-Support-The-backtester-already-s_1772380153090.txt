JavaScriptfunction parseTimeframe(tf) {
  const map = {
    '1min':   { multiplier: 1,   timespan: 'minute' },
    '5min':   { multiplier: 5,   timespan: 'minute' },
    '15min':  { multiplier: 15,  timespan: 'minute' },
    '1hour':  { multiplier: 1,   timespan: 'hour'   },
    '4hour':  { multiplier: 4,   timespan: 'hour'   },
    'daily':  { multiplier: 1,   timespan: 'day'    }
  };
  return map[tf] || map['daily'];
}

Manual stresses confluence of multiple concepts (price, volume, candlesticks, MAs, trends) for edge. Your 0–11pt system already captures this, but we can add explicit "convergence bonus" when ≥2 patterns align on same bar (e.g., Wedge + 3 Bar Play or Buy Setup + Retest).Paste into trader.ts (signal processing / emitTradeSignal):

JavaScript// Near emitTradeSignal() or after pattern detection
const sameBarSignals = getSignalsOnBar(signal.timestamp, signal.symbol); // helper: return array of patterns on same timestamp

let convergenceBonus = 0;
if (sameBarSignals.length >= 2) {
  convergenceBonus = (sameBarSignals.length - 1) * 1.5; // +1.5pt per extra pattern
  signal.confluence += convergenceBonus;
  signal.notes = `Multi-pattern convergence: ${sameBarSignals.join(', ')} (+${convergenceBonus}pt)`;
  console.log(`[CONVERGENCE BOOST] ${signal.symbol} @ ${signal.timestamp}: ${signal.notes}`);
}

// Helper function (add to scanner scope)
function getSignalsOnBar(timestamp, symbol) {
  // In-memory buffer or query recent signals
  return recentSignalsBuffer
    .filter(s => s.timestamp === timestamp && s.symbol === symbol)
    .map(s => s.pattern);
}
