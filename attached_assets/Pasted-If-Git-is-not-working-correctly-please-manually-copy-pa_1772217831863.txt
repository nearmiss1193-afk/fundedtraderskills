import { createHmac } from "crypto";

interface CrossTradePayload {

export async function sendToCrossTrade(params: {
  symbol: string;
  direction: string;
  orderType: string;
  account?: string;
}) {
  const WEBHOOK_URL = process.env.CROSSTRADE_WEBHOOK_URL;
  const CROSSTRADE_KEY = process.env.CROSSTRADE_KEY;
  const DEFAULT_ACCOUNT = process.env.CROSSTRADE_ACCOUNT || "SIM101";

  if (!WEBHOOK_URL || !CROSSTRADE_KEY) {
    console.error("❌ CrossTrade missing WEBHOOK_URL or KEY");
    return { success: false, message: "CrossTrade configuration missing" };
  }

  // Use provided account or fall back to default
  const targetAccount = params.account || DEFAULT_ACCOUNT;

  // SAFETY GUARDRAIL: Only allow SIM accounts unless explicitly configured
  // This prevents accidental live trades if not intended.
  if (!targetAccount.toUpperCase().startsWith("SIM") && process.env.ALLOW_LIVE_TRADES !== "true") {
    // If you want to trade on live accounts, add ALLOW_LIVE_TRADES=true to Replit Secrets
    return { 
      success: false, 
      message: `Safety Block: Account '${targetAccount}' is NOT a SIM account. Add 'ALLOW_LIVE_TRADES=true' to Secrets to override.` 
    };
  }

  const payload = {
    symbol: params.symbol,
    action: params.direction.toUpperCase(), // BUY or SELL for CrossTrade
    orderType: params.orderType,
    quantity: parseInt(process.env.MAX_CONTRACTS || "1"),
    account: targetAccount
  };

  // CrossTrade typically expects a simple text/plain payload for many webhooks,
  // or a formatted JSON string depending on the endpoint.
  // Using their standard text format: SYMBOL ACTION QTY TYPE ACCOUNT
  const textPayload = `${payload.symbol} ${payload.action} ${payload.quantity} ${payload.orderType} ${payload.account}`;

  console.log(`🚀 Sending to CrossTrade [${targetAccount}]: ${textPayload}`);

  try {
    const response = await fetch(WEBHOOK_URL, {
      method: 'POST',
      headers: {
        'Content-Type': 'text/plain',
        'x-crosstrade-key': CROSSTRADE_KEY
      },
      body: textPayload
    });

    if (response.ok) {
      return { success: true, message: `Order sent to ${targetAccount}` };
    } else {
      const err = await response.text();
      return { success: false, message: `CrossTrade Error: ${err}` };
    }
  } catch (error) {
    return { success: false, message: `Network Error: ${error}` };
  }
}

## 3. Replit Secrets (Environment Variables)

Ensure these are set in your Replit **Tools -> Secrets**:

1. `CROSSTRADE_WEBHOOK_URL`: (Your unique URL)
2. `CROSSTRADE_KEY`: (Your CrossTrade Key)
3. `CROSSTRADE_ACCOUNT`: `SIM101` (Default)
4. `ALLOW_LIVE_TRADES`: `true` (Only if you are switching to a real funded account)
