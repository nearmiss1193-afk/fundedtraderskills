JavaScript// Bull Flag Pullback (New High-Prob Setup)
function detectBullFlagPullback(bars, symbol, timeframe) {
  if (bars.length < 10) return null;

  const recent = bars.slice(-10);
  const ignitingBar = recent[0]; // strong up bar
  if (ignitingBar.close - ignitingBar.open < averageRange(recent.slice(0, -5)) * 1.5 || ignitingBar.volume < averageVolume(recent.slice(0, -5)) * 2) return null;

  const pullbackBars = recent.slice(1, -1); // 3–5 pullback bars
  if (pullbackBars.length < 3 || pullbackBars.length > 5) return null;

  const flagHigh = Math.max(...pullbackBars.map(b => b.high));
  const lastBar = recent[recent.length - 1];
  const breakout = lastBar.close > flagHigh && lastBar.volume > averageVolume(recent.slice(-5)) * 1.5;

  if (!breakout) return null;

  const confluence = calculateConfluence({
    volSurge: true,
    volType: getVolumeType(lastBar, recent),
    tailPresent: hasTail(lastBar),
    strongBody: isStrongBody(lastBar),
    maRespect: lastBar.close > ema(recent.map(b => b.close), 21)[recent.length - 1],
    htfAligned: isHTFAligned(symbol, 'long'),
    gapLevel: detectGapLevel(recent),
    wBottomTop: detectWBottomTop(recent, 'long'),
    vpConfluence: getVolumeProfileConfluence(recent, lastBar.close),
    ofConfluence: getOrderFlowConfluence(recent)
  });

  if (confluence < 8) return null;

  const entry = lastBar.close;
  const stop = Math.min(...pullbackBars.map(b => b.low));
  const target = entry + (entry - stop) * 2; // 1:2 R:R

  return {
    pattern: 'bullflagpullback',
    direction: 'long',
    entry,
    stop,
    target,
    confluence,
    volType: getVolumeType(lastBar, recent),
    timestamp: lastBar.timestamp,
    notes: `Pullback bars: ${pullbackBars.length}, Flag high: ${flagHigh.toFixed(2)}`
  };
}
