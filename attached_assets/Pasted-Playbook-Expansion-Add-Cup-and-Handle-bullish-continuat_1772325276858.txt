def detect_rising_wedge(df, window=30):
    recent = df.tail(window)
    highs = recent['high'].values
    lows = recent['low'].values
    if len(highs) < 10: return False
    # Simple slope check: highs slope up, lows slope up but steeper convergence
    high_slope = np.polyfit(range(len(highs)), highs, 1)[0]
    low_slope = np.polyfit(range(len(lows)), lows, 1)[0]
    vol_decline = recent['volume'].mean() < df['volume_avg'].mean() * 1.2
    return high_slope > 0 and low_slope > 0 and abs(high_slope - low_slope) < 0.01 and vol_decline  # Converging + vol drop
