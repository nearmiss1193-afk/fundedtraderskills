Run Multi-Symbol Edge Scan (Prompt to Paste in Replit)
Paste this into the Replit console (after server is running) or create a temporary file multi-scan-test.js and run it with node multi-scan-test.js. It scans the full Apex-compatible list (high-liquidity first), 5-min timeframe, min 7pt confluence. Adjust timeframe/minConfluence as needed after first run.

JavaScript// multi-scan-test.js - Paste & run in Replit console or as file
const axios = require('axios');

const BASE_URL = 'http://localhost:3000'; // change to your Replit URL if external

const symbols = [
  'ES', 'MES', 'NQ', 'MNQ', 'YM', 'MYM', 'RTY', 'M2K',
  'CL', 'MCL', 'GC', 'MGC', 'SI', 'HG',
  'BTC', 'ETH', 'ZB', 'ZN', 'ZC', 'ZS', 'ZW'
];

const params = {
  timeframe: '5min',              // try '15min', '60min' after
  startDate: '2020-01-01',
  endDate: '2026-02-28',
  patterns: ['3bar', '4bar', 'buysetup', 'retest', 'breakout', 'climax'],
  minConfluence: 7,
  riskPerTrade: 1                 // contracts
};

async function runMultiScan() {
  console.log('Starting multi-symbol edge scan...');
  console.log('Symbols:', symbols.join(', '));
  console.log('Timeframe:', params.timeframe);
  console.log('Min Confluence:', params.minConfluence);

  try {
    const response = await axios.post(`${BASE_URL}/api/backtest/multi`, {
      symbols,
      ...params
    });

    const data = response.data;

    console.log('\n=== Multi-Symbol Edge Scan Summary ===');
    console.log(`Symbols Scanned: ${data.summary.symbolsScanned}`);
    console.log(`Total Trades: ${data.summary.totalTrades}`);
    console.log(`Overall Win Rate: ${(data.summary.winRate * 100).toFixed(2)}%`);
    console.log(`Overall Profit Factor: ${data.summary.profitFactor.toFixed(2)}`);
    console.log(`Net P&L: ${data.summary.netPnL.toFixed(2)}`);
    console.log(`Max Drawdown: ${data.summary.maxDrawdown.toFixed(2)}%`);

    console.log('\n=== Top 5 Symbols by P&L ===');
    data.results
      .sort((a, b) => b.totalPnL - a.totalPnL)
      .slice(0, 5)
      .forEach(r => {
        console.log(`${r.symbol.padEnd(6)} | P&L: ${r.totalPnL.toFixed(2).padStart(8)} | Win%: ${(r.winRate * 100).toFixed(1)} | PF: ${r.profitFactor.toFixed(2)} | Trades: ${r.tradeCount}`);
      });

    console.log('\n=== Top Patterns Across All Symbols ===');
    // Assuming endpoint returns patternBreakdown (add if missing)
    if (data.patternBreakdown) {
      Object.entries(data.patternBreakdown)
        .sort(([,a], [,b]) => b.pnl - a.pnl)
        .slice(0, 5)
        .forEach(([pat, stats]) => {
          console.log(`${pat.padEnd(12)} | P&L: ${stats.pnl.toFixed(2)} | Win%: ${(stats.winRate * 100).toFixed(1)} | Trades: ${stats.count}`);
        });
    } else {
      console.log('(Pattern breakdown not returned - check endpoint)');
    }

    console.log('\nNext: Open Edge Builder UI → Export CSV for full analysis');

  } catch (err) {
    console.error('Scan failed:', err.message);
    if (err.response) console.error('Response:', err.response.data);
  }
}

runMultiScan();

Add Edge Heatmap Tab to Dashboard (Prompt for Replit / Vercel)
Paste this into your frontend code (likely src/components/EdgeBuilder.jsx or similar). Adds a simple heatmap tab showing symbol × timeframe × pattern performance.

jsx// Add to EdgeBuilder.jsx or new EdgeHeatmap.jsx
import React, { useState, useEffect } from 'react';
import { AgGridReact } from 'ag-grid-react';
import 'ag-grid-community/styles/ag-grid.css';
import 'ag-grid-community/styles/ag-theme-alpine.css';

const EdgeHeatmap = ({ lastMultiResults }) => {
  const [rowData, setRowData] = useState([]);

  useEffect(() => {
    if (!lastMultiResults?.results) return;

    const heatData = [];
    lastMultiResults.results.forEach(sym => {
      if (sym.timeframeBreakdown) {
        Object.entries(sym.timeframeBreakdown).forEach(([tf, tfData]) => {
          Object.entries(tfData.patterns || {}).forEach(([pat, stats]) => {
            heatData.push({
              symbol: sym.symbol,
              timeframe: tf,
              pattern: pat,
              winRate: (stats.winRate * 100).toFixed(1) + '%',
              pf: stats.profitFactor.toFixed(2),
              trades: stats.count,
              pnl: stats.pnl.toFixed(2),
              expectancy: stats.expectancy?.toFixed(2) || 'N/A'
            });
          });
        });
      }
    });

    setRowData(heatData);
  }, [lastMultiResults]);

  const columnDefs = [
    { headerName: 'Symbol', field: 'symbol', width: 100 },
    { headerName: 'Timeframe', field: 'timeframe', width: 120 },
    { headerName: 'Pattern', field: 'pattern', width: 140 },
    { headerName: 'Win %', field: 'winRate', width: 100 },
    { headerName: 'PF', field: 'pf', width: 80 },
    { headerName: 'Trades', field: 'trades', width: 90 },
    { headerName: 'P&L', field: 'pnl', width: 110 },
    { headerName: 'Expectancy', field: 'expectancy', width: 110 }
  ];

  return (
    <div style={{ height: '600px', width: '100%' }} className="ag-theme-alpine">
      <h3>Edge Heatmap (Symbol × Timeframe × Pattern)</h3>
      <AgGridReact
        rowData={rowData}
        columnDefs={columnDefs}
        defaultColDef={{ sortable: true, filter: true, resizable: true }}
      />
    </div>
  );
};

export default EdgeHeatmap;
