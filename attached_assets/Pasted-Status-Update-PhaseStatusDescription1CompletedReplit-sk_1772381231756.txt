Head and Shoulders + Inverse Head and Shoulders Patterns Added
The classic Head and Shoulders (bearish reversal) and Inverse Head and Shoulders (bullish reversal) patterns are now integrated into both the backtester and live scanner.Pattern Rules (aligned with standard technical analysis + Live Traders confluence style):
Head and Shoulders (bearish): Left shoulder → higher head → right shoulder (similar height to left), neckline connecting lows, breakdown below neckline with volume surge.
Inverse Head and Shoulders (bullish): Left shoulder → lower head → right shoulder, neckline connecting highs, breakout above neckline with volume surge.
Lookback: 30–100 bars (typically 3–12 months on daily, shorter on intraday)
Confluence boost: +1 for volume increase on breakdown/breakout, +1 for igniting volume, +1 for MTF alignment, +1 for gap near neckline, +1 for tail on breakout bar.
Risk: SL above right shoulder high (bearish) or below right shoulder low (bullish); target = head height projected from neckline.

JavaScript// Head and Shoulders + Inverse - add to detectPatterns() or as standalone functions

function detectHeadAndShoulders(bars) { // Bearish
  if (bars.length < 50) return null;

  const recent = bars.slice(-100);
  // Find peaks (shoulders & head)
  const peaks = [];
  for (let i = 5; i < recent.length - 5; i++) {
    if (recent[i].high > recent[i-5].high && recent[i].high > recent[i+5].high) {
      peaks.push({ idx: i, high: recent[i].high });
    }
  }

  if (peaks.length < 3) return null;

  // Last three peaks: left shoulder, head, right shoulder
  const ls = peaks[peaks.length - 3];
  const head = peaks[peaks.length - 2];
  const rs = peaks[peaks.length - 1];

  const isHAndS = 
    head.high > ls.high && head.high > rs.high && 
    Math.abs(ls.high - rs.high) / head.high < 0.08; // shoulders similar height (±8%)

  // Neckline: connect lows between shoulders and head
  const leftLow = Math.min(...recent.slice(ls.idx, head.idx).map(b => b.low));
  const rightLow = Math.min(...recent.slice(head.idx, rs.idx).map(b => b.low));
  const necklineSlope = (rightLow - leftLow) / (rs.idx - ls.idx);
  const currentNeck = leftLow + necklineSlope * (recent.length - 1 - ls.idx);

  const lastBar = recent[recent.length - 1];
  const breakdown = lastBar.close < currentNeck && lastBar.volume > 1.5 * averageVolume(recent.slice(-10));

  if (!isHAndS || !breakdown) return null;

  const confluence = calculateConfluence({
    volSurge: true,
    volType: getVolumeType(lastBar, recent),
    tailPresent: hasTail(lastBar),
    strongBody: isStrongBody(lastBar),
    maRespect: lastBar.close < ema(recent.map(b => b.close), 21)[recent.length - 1],
    htfAligned: isHTFAligned(symbol, 'short'),
    gapLevel: detectGapLevel(recent),
    wBottomTop: detectWBottomTop(recent, 'short')
  });

  if (confluence < 8) return null;

  const entry = lastBar.close;
  const stop = Math.max(ls.high, rs.high, head.high) * 1.01; // above right shoulder
  const target = entry - (head.high - currentNeck); // head height projected down

  return {
    pattern: 'headandshoulders',
    direction: 'short',
    entry,
    stop,
    target,
    confluence,
    volType: getVolumeType(lastBar, recent),
    timestamp: lastBar.timestamp,
    notes: `Head height: ${(head.high - currentNeck).toFixed(2)}, Neckline: ${currentNeck.toFixed(2)}`
  };
}
