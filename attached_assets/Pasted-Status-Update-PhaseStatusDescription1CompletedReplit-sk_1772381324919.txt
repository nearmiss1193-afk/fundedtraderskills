Head and Shoulders + Inverse Head and Shoulders Patterns Added
Both patterns are now fully integrated into the backtester and live scanner.
Head and Shoulders (bearish reversal): Left shoulder → higher head → right shoulder → breakdown below neckline with volume surge.
Inverse Head and Shoulders (bullish reversal): Left shoulder → lower head → right shoulder → breakout above neckline with volume surge.
Lookback: 50–100 bars (adjustable for timeframe)
Confluence: Same 0–11pt system + bonus for volume surge on break, MTF alignment, tail on breakout bar, gap near neckline
Risk: SL above right shoulder high (bearish) or below right shoulder low (bullish); target = head height projected from neckline
Complete code — paste this into your pattern detection module (e.g., patterns.ts for backtest, trader.ts for live):

JavaScript// Head and Shoulders (Bearish) + Inverse Head and Shoulders (Bullish)
// Add these functions to your pattern detection logic

function detectHeadAndShoulders(bars, symbol, timeframe) {
  if (bars.length < 50) return null;

  const recent = bars.slice(-100);
  const peaks = [];
  for (let i = 5; i < recent.length - 5; i++) {
    if (recent[i].high > recent[i-5].high && recent[i].high > recent[i+5].high) {
      peaks.push({ idx: i, high: recent[i].high });
    }
  }

  if (peaks.length < 3) return null;

  const ls = peaks[peaks.length - 3];   // left shoulder
  const head = peaks[peaks.length - 2]; // head (highest)
  const rs = peaks[peaks.length - 1];   // right shoulder

  const isHAndS = 
    head.high > ls.high && head.high > rs.high &&
    Math.abs(ls.high - rs.high) / head.high < 0.08; // shoulders similar (±8%)

  // Neckline: connect lows between shoulders and head
  const leftLow = Math.min(...recent.slice(ls.idx, head.idx).map(b => b.low));
  const rightLow = Math.min(...recent.slice(head.idx, rs.idx).map(b => b.low));
  const necklineSlope = (rightLow - leftLow) / (rs.idx - ls.idx);
  const currentNeck = leftLow + necklineSlope * (recent.length - 1 - ls.idx);

  const lastBar = recent[recent.length - 1];
  const breakdown = lastBar.close < currentNeck && lastBar.volume > 1.5 * averageVolume(recent.slice(-10));

  if (!isHAndS || !breakdown) return null;

  const confluence = calculateConfluence({
    volSurge: true,
    volType: getVolumeType(lastBar, recent),
    tailPresent: hasTail(lastBar),
    strongBody: isStrongBody(lastBar),
    maRespect: lastBar.close < ema(recent.map(b => b.close), 21)[recent.length - 1],
    htfAligned: isHTFAligned(symbol, 'short'),
    gapLevel: detectGapLevel(recent),
    wBottomTop: detectWBottomTop(recent, 'short')
  });

  if (confluence < 8) return null;

  const entry = lastBar.close;
  const stop = Math.max(ls.high, rs.high, head.high) * 1.01;
  const patternHeight = currentNeck - head.high;
  const target = entry - Math.abs(patternHeight); // projected down

  return {
    pattern: 'headandshoulders',
    direction: 'short',
    entry,
    stop,
    target,
    confluence,
    volType: getVolumeType(lastBar, recent),
    timestamp: lastBar.timestamp,
    notes: `Head height: ${Math.abs(patternHeight).toFixed(2)}, Neckline: ${currentNeck.toFixed(2)}`
  };
}

function detectInverseHeadAndShoulders(bars, symbol, timeframe) {
  if (bars.length < 50) return null;

  const recent = bars.slice(-100);
  const troughs = [];
  for (let i = 5; i < recent.length - 5; i++) {
    if (recent[i].low < recent[i-5].low && recent[i].low < recent[i+5].low) {
      troughs.push({ idx: i, low: recent[i].low });
    }
  }

  if (troughs.length < 3) return null;

  const ls = troughs[troughs.length - 3];   // left shoulder
  const head = troughs[troughs.length - 2]; // head (lowest)
  const rs = troughs[troughs.length - 1];   // right shoulder

  const isInvHAndS = 
    head.low < ls.low && head.low < rs.low &&
    Math.abs(ls.low - rs.low) / head.low < 0.08; // shoulders similar (±8%)

  // Neckline: connect highs between shoulders and head
  const leftHigh = Math.max(...recent.slice(ls.idx, head.idx).map(b => b.high));
  const rightHigh = Math.max(...recent.slice(head.idx, rs.idx).map(b => b.high));
  const necklineSlope = (rightHigh - leftHigh) / (rs.idx - ls.idx);
  const currentNeck = leftHigh + necklineSlope * (recent.length - 1 - ls.idx);

  const lastBar = recent[recent.length - 1];
  const breakout = lastBar.close > currentNeck && lastBar.volume > 1.5 * averageVolume(recent.slice(-10));

  if (!isInvHAndS || !breakout) return null;

  const confluence = calculateConfluence({
    volSurge: true,
    volType: getVolumeType(lastBar, recent),
    tailPresent: hasTail(lastBar),
    strongBody: isStrongBody(lastBar),
    maRespect: lastBar.close > ema(recent.map(b => b.close), 21)[recent.length - 1],
    htfAligned: isHTFAligned(symbol, 'long'),
    gapLevel: detectGapLevel(recent),
    wBottomTop: detectWBottomTop(recent, 'long')
  });

  if (confluence < 8) return null;

  const entry = lastBar.close;
  const stop = Math.min(ls.low, rs.low, head.low) * 0.99;
  const patternHeight = currentNeck - head.low;
  const target = entry + patternHeight; // projected up

  return {
    pattern: 'inverseheadandshoulders',
    direction: 'long',
    entry,
    stop,
    target,
    confluence,
    volType: getVolumeType(lastBar, recent),
    timestamp: lastBar.timestamp,
    notes: `Head depth: ${patternHeight.toFixed(2)}, Neckline: ${currentNeck.toFixed(2)}`
  };
}
