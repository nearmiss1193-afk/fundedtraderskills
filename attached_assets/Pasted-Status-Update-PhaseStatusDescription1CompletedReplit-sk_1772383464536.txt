JavaScriptfunction parseTimeframe(tf) {
  const map = {
    '1min':   { multiplier: 1, timespan: 'minute' },
    '2min':   { multiplier: 2, timespan: 'minute' },
    '3min':   { multiplier: 3, timespan: 'minute' },
    '5min':   { multiplier: 5, timespan: 'minute' },
    '15min':  { multiplier: 15, timespan: 'minute' },
    '1hour':  { multiplier: 1, timespan: 'hour' },
    '4hour':  { multiplier: 4, timespan: 'hour' },
    'daily':  { multiplier: 1, timespan: 'day' }
  };
  return map[tf] || map['5min']; // default to 5min
}

Multiple Timeframes in Multi-Scanner
Update multi-scan endpoint in server/routes.ts to accept array of timeframes (e.g., ["2min", "3min", "5min"]), run parallel, and detect multi-TF alignment (signals on same bar across TFs add convergence bonus).

JavaScriptapp.post('/api/backtest/multi', async (req, res) => {
  const { symbols, timeframes = ['5min'], patterns, minConfluence, startDate, endDate } = req.body;
  const results = [];
  for (const tf of timeframes) {
    for (const sym of symbols) {
      const bars = await fetchPolygonAggs(sym, tf, startDate, endDate);
      const signals = detectAllPatterns(bars, sym, tf);
      results.push({ sym, tf, signals: signals.filter(s => s.confluence >= minConfluence) });
    }
  }
  // Check multi-TF alignment
  results.forEach(r => {
    r.multiTfAligned = checkMultiTfConvergence(r.signals, results, r.sym, r.tf);
  });
  res.json({ results });
}

// Helper: check if signal aligns across TFs (add bonus)
function checkMultiTfConvergence(signals, allResults, sym, tf) {
  let bonus = 0;
  signals.forEach(s => {
    allResults.filter(res => res.sym === sym && res.tf !== tf).forEach(res => {
      const matching = res.signals.find(ms => ms.timestamp === s.timestamp && ms.pattern === s.pattern);
      if (matching) {
        bonus += 1.5; // +1.5pt per aligned TF
        s.confluence += bonus;
        s.notes += `Multi-TF alignment with ${res.tf} (+${bonus}pt)`;
      }
    });
  });
  return bonus > 0;
}
