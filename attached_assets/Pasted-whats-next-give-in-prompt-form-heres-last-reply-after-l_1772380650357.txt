Add Edge Heatmap Tab to Dashboard (Prompt for Replit / Vercel)
Paste this into your frontend code (likely src/components/EdgeBuilder.jsx or similar). Adds a simple heatmap tab showing symbol × timeframe × pattern performance.

jsx// Add to EdgeBuilder.jsx or new EdgeHeatmap.jsx
import React, { useState, useEffect } from 'react';
import { AgGridReact } from 'ag-grid-react';
import 'ag-grid-community/styles/ag-grid.css';
import 'ag-grid-community/styles/ag-theme-alpine.css';

const EdgeHeatmap = ({ lastMultiResults }) => {
  const [rowData, setRowData] = useState([]);

  useEffect(() => {
    if (!lastMultiResults?.results) return;

    const heatData = [];
    lastMultiResults.results.forEach(sym => {
      if (sym.timeframeBreakdown) {
        Object.entries(sym.timeframeBreakdown).forEach(([tf, tfData]) => {
          Object.entries(tfData.patterns || {}).forEach(([pat, stats]) => {
            heatData.push({
              symbol: sym.symbol,
              timeframe: tf,
              pattern: pat,
              winRate: (stats.winRate * 100).toFixed(1) + '%',
              pf: stats.profitFactor.toFixed(2),
              trades: stats.count,
              pnl: stats.pnl.toFixed(2),
              expectancy: stats.expectancy?.toFixed(2) || 'N/A'
            });
          });
        });
      }
    });

    setRowData(heatData);
  }, [lastMultiResults]);

  const columnDefs = [
    { headerName: 'Symbol', field: 'symbol', width: 100 },
    { headerName: 'Timeframe', field: 'timeframe', width: 120 },
    { headerName: 'Pattern', field: 'pattern', width: 140 },
    { headerName: 'Win %', field: 'winRate', width: 100 },
    { headerName: 'PF', field: 'pf', width: 80 },
    { headerName: 'Trades', field: 'trades', width: 90 },
    { headerName: 'P&L', field: 'pnl', width: 110 },
    { headerName: 'Expectancy', field: 'expectancy', width: 110 }
  ];

  return (
    <div style={{ height: '600px', width: '100%' }} className="ag-theme-alpine">
      <h3>Edge Heatmap (Symbol × Timeframe × Pattern)</h3>
      <AgGridReact
        rowData={rowData}
        columnDefs={columnDefs}
        defaultColDef={{ sortable: true, filter: true, resizable: true }}
      />
    </div>
  );
};

export default EdgeHeatmap;

const CACHE = new Map(); // {key: {data: bars, timestamp: Date}}
const RATE_LIMIT = 5; // calls/min
let callsThisMinute = 0;
let lastMinute = Date.now();

async function fetchPolygonAggs(symbol, timeframe, start, end, retryCount = 0) {
  const cacheKey = `${symbol}-${timeframe}-${start}-${end}`;
  const cached = CACHE.get(cacheKey);
  if (cached && (Date.now() - cached.timestamp) < 300000) { // 5 min TTL
    return cached.data;
  }

  // Rate guard: pause if near limit
  if (Date.now() - lastMinute > 60000) {
    callsThisMinute = 0;
    lastMinute = Date.now();
  }
  if (callsThisMinute >= RATE_LIMIT) {
    await new Promise(resolve => setTimeout(resolve, 60000 - (Date.now() - lastMinute)));
    callsThisMinute = 0;
    lastMinute = Date.now();
  }
  callsThisMinute++;

  try {
    const response = await fetch(`https://api.polygon.io/v2/aggs/ticker/${symbol}/range/${timeframe}/1/${start}/${end}?apiKey=${POLYGON_API_KEY}`);
    if (response.status === 429) {
      if (retryCount >= 3) throw new Error('Max retries exceeded');
      const backoff = 30000 + Math.random() * 35000; // 30-65s
      await new Promise(resolve => setTimeout(resolve, backoff));
      return fetchPolygonAggs(symbol, timeframe, start, end, retryCount + 1);
    }
    if (!response.ok) return []; // Silent [] on non-200
    const data = await response.json();
    const bars = data.results.map(r => ({ time: r.t, open: r.o, high: r.h, low: r.l, close: r.c, volume: r.v }));
    CACHE.set(cacheKey, { data: bars, timestamp: Date.now() });
    return bars;
  } catch (err) {
    console.error('Fetch error:', err);
    return [];
  }
}

JavaScript// Add to pattern detection functions (e.g., inside detectPatterns() or as new function detectWedge())
function detectWedge(bars, symbol, timeframe) {
  const recent = bars.slice(-30); // lookback 30 bars for wedge formation
  if (recent.length < 15) return null;

  // Simple wedge detection: converging highs/lows
  const highs = recent.map(b => b.high);
  const lows = recent.map(b => b.low);

  const highSlope = linearRegressionSlope(highs);   // helper function below
  const lowSlope = linearRegressionSlope(lows);

  const isRising = highSlope > 0 && lowSlope > 0 && Math.abs(highSlope - lowSlope) < 0.0005; // converging upward
  const isFalling = highSlope < 0 && lowSlope < 0 && Math.abs(highSlope - lowSlope) < 0.0005; // converging downward

  if (!isRising && !isFalling) return null;

  // Breakout confirmation: last bar breaks upper (falling wedge) or lower (rising wedge) trendline
  const last = recent[recent.length - 1];
  const prev = recent[recent.length - 2];

  const volSurge = last.volume > 1.5 * averageVolume(recent.slice(-10));
  const aboveEMA = last.close > ema(recent.map(b => b.close), 21)[recent.length - 1];

  let direction = null;
  let entryPrice = last.close;
  let stopPrice = 0;
  let targetPrice = 0;

  if (isFalling && last.close > prev.high && volSurge && aboveEMA) {
    // Bullish falling wedge breakout
    direction = 'long';
    stopPrice = Math.min(...recent.slice(-5).map(b => b.low)); // below recent lows
    targetPrice = entryPrice + (entryPrice - stopPrice) * 2; // 1:2 R:R
  } else if (isRising && last.close < prev.low && volSurge) {
    // Bearish rising wedge breakdown
    direction = 'short';
    stopPrice = Math.max(...recent.slice(-5).map(b => b.high));
    targetPrice = entryPrice - (stopPrice - entryPrice) * 2;
  }

  if (!direction) return null;

  const confluence = calculateConfluence({
    volSurge,
    volType: getVolumeType(last, recent),
    tailPresent: hasTail(last),
    strongBody: isStrongBody(last),
    maRespect: aboveEMA,
    htfAligned: isHTFAligned(symbol, direction),
    gapLevel: detectGapLevel(recent),
    wBottomTop: detectWBottomTop(recent, direction)
  });

  if (confluence < 7) return null; // enforce live threshold

  return {
    symbol,
    timeframe,
    pattern: 'wedge',
    direction,
    entry: entryPrice,
    stop: stopPrice,
    target: targetPrice,
    confluence,
    volType: getVolumeType(last, recent),
    timestamp: last.timestamp
  };
}

// Helper: simple linear regression slope
function linearRegressionSlope(values) {
  const n = values.length;
  const x = Array.from({length: n}, (_, i) => i);
  const sumX = x.reduce((a, b) => a + b, 0);
  const sumY = values.reduce((a, b) => a + b, 0);
  const sumXY = x.reduce((sum, xi, i) => sum + xi * values[i], 0);
  const sumX2 = x.reduce((sum, xi) => sum + xi * xi, 0);
  return (n * sumXY - sumX * sumY) / (n * sumX2 - sumX * sumX);
}

Add Wedge Detection to Live Scanner
To capture the high-edge Wedge setups (especially NQ/Wedge) in real-time, paste the following into your live scanner file (likely trader.ts or main pattern detection module). This adds the same Wedge logic used in backtester (rising/falling convergence → breakout with volume + trend filter).

JavaScript// Add to pattern detection section (e.g., inside scan loop or detectPatterns())
function detectWedge(bars) {
  if (bars.length < 20) return null;

  const recent = bars.slice(-30);
  const highs = recent.map(b => b.high);
  const lows = recent.map(b => b.low);

  const highSlope = linearRegressionSlope(highs);
  const lowSlope = linearRegressionSlope(lows);

  const isRisingWedge = highSlope > 0 && lowSlope > 0 && Math.abs(highSlope - lowSlope) < 0.0008;
  const isFallingWedge = highSlope < 0 && lowSlope < 0 && Math.abs(highSlope - lowSlope) < 0.0008;

  if (!isRisingWedge && !isFallingWedge) return null;

  const lastBar = recent[recent.length - 1];
  const prevBar = recent[recent.length - 2];

  const volSurge = lastBar.volume > 1.5 * averageVolume(recent.slice(-10));
  const aboveEMA21 = lastBar.close > ema(recent.map(b => b.close), 21)[recent.length - 1];

  let signal = null;

  if (isFallingWedge && lastBar.close > prevBar.high && volSurge && aboveEMA21) {
    // Bullish breakout from falling wedge
    signal = {
      pattern: 'wedge',
      direction: 'long',
      entry: lastBar.close,
      stop: Math.min(...recent.slice(-5).map(b => b.low)),
      target: lastBar.close + (lastBar.close - Math.min(...recent.slice(-5).map(b => b.low))) * 2,
      confluence: calculateConfluence({ volSurge, volType: getVolumeType(lastBar, recent), ... }),
      volType: getVolumeType(lastBar, recent),
      timestamp: lastBar.timestamp
    };
  } else if (isRisingWedge && lastBar.close < prevBar.low && volSurge) {
    // Bearish breakdown from rising wedge
    signal = {
      pattern: 'wedge',
      direction: 'short',
      entry: lastBar.close,
      stop: Math.max(...recent.slice(-5).map(b => b.high)),
      target: lastBar.close - (Math.max(...recent.slice(-5).map(b => b.high)) - lastBar.close) * 2,
      confluence: calculateConfluence({ volSurge, volType: getVolumeType(lastBar, recent), ... }),
      volType: getVolumeType(lastBar, recent),
      timestamp: lastBar.timestamp
    };
  }

  if (signal && signal.confluence >= 8) {
    return signal;
  }
  return null;
}
